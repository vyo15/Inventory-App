& {
  $ErrorActionPreference = "Stop"

  $ProjectPath = "C:\Users\vio15\Project\Inventory-App"
  $OutputDirectory = "C:\Users\vio15\Project"
  $BaseName = "Inventory-App"

  function Assert-NativeSuccess {
    param([string]$Step)

    if ($LASTEXITCODE -ne 0) {
      throw "$Step gagal dengan exit code $LASTEXITCODE."
    }
  }

  function Test-ExcludedSourcePath {
    param([string]$RelativePath)

    $Normalized = $RelativePath.Replace("\", "/").TrimStart([char[]]@("/"))

    if ([string]::IsNullOrWhiteSpace($Normalized)) {
      return $true
    }

    # Hanya root runtime data dan root backups yang dikecualikan.
    # frontend/src/data tetap wajib ikut.
    if ($Normalized -match "^(data|backups)(/|$)") {
      return $true
    }

    $ExcludedSegments = @(
      ".git",
      "node_modules",
      "dist",
      "dist-ssr",
      "build",
      "coverage",
      ".cache",
      ".vite",
      ".firebase"
    )

    foreach ($Segment in $Normalized.Split("/")) {
      if ($ExcludedSegments -contains $Segment) {
        return $true
      }
    }

    if ($Normalized -match "^frontend/frontend(/|$)") {
      return $true
    }

    $FileName = [System.IO.Path]::GetFileName($Normalized)

    if ($FileName -eq ".env") {
      return $true
    }

    if (
      $FileName.EndsWith(
        ".local",
        [System.StringComparison]::OrdinalIgnoreCase
      )
    ) {
      return $true
    }

    if (
      $Normalized -match
      "(?i)(^|/)_IMS_(DELETE_FILES|PATCH_INFO)\.txt$"
    ) {
      return $true
    }

    if ($Normalized -match "(?i)\.(log|tmp|bak|zip|rar|7z)$") {
      return $true
    }

    if ($Normalized -match "(?i)\.sqlite(?:-wal|-shm)?$") {
      return $true
    }

    if ($Normalized -match "(?i)\.imsbackup$") {
      return $true
    }

    if ($Normalized -match "(?i)\.imsbackup\.manifest\.json$") {
      return $true
    }

    if ($Normalized -match "(?i)\.imsbak\.zip$") {
      return $true
    }

    return $false
  }

  if (-not (Test-Path -LiteralPath $ProjectPath -PathType Container)) {
    throw "Folder project tidak ditemukan: $ProjectPath"
  }

  Set-Location $ProjectPath

  $GitRoot = [string](git rev-parse --show-toplevel)
  Assert-NativeSuccess "Membaca root repository Git"
  $GitRoot = $GitRoot.Trim()

  if ([string]::IsNullOrWhiteSpace($GitRoot)) {
    throw "Root repository Git tidak ditemukan."
  }

  Set-Location $GitRoot

  $Branch = [string](git branch --show-current)
  Assert-NativeSuccess "Membaca branch Git"
  $Branch = $Branch.Trim()

  if ([string]::IsNullOrWhiteSpace($Branch)) {
    $Branch = "detached"
  }

  $Branch = $Branch -replace "[^A-Za-z0-9._-]", "-"

  $Commit = [string](git rev-parse --short=8 HEAD)
  Assert-NativeSuccess "Membaca commit Git"
  $Commit = $Commit.Trim()

  $GitChanges = @(git status --porcelain)
  Assert-NativeSuccess "Membaca status Git"

  $Status = "clean"

  if ($GitChanges.Count -gt 0) {
    $Status = "dirty"
  }

  $Timestamp = Get-Date -Format "yyyyMMdd-HHmmssfff"
  $OutputFileName = "$BaseName-$Timestamp-$Branch-$Commit-$Status.zip"
  $OutputZip = Join-Path -Path $OutputDirectory -ChildPath $OutputFileName

  New-Item -ItemType Directory -Path $OutputDirectory -Force | Out-Null
  Remove-Item -LiteralPath $OutputZip -Force -ErrorAction SilentlyContinue

  Write-Host ""
  Write-Host "Membaca source terbaru dari working tree Git..."
  Write-Host "Root   : $GitRoot"
  Write-Host "Branch : $Branch"
  Write-Host "Commit : $Commit"
  Write-Host "Status : $Status"
  Write-Host "Output : $OutputZip"

  if ($Status -eq "dirty") {
    Write-Host ""
    Write-Warning "Perubahan lokal yang belum di-commit akan ikut ke ZIP."
  }

  $GitFiles = @(
    git -c core.quotepath=false ls-files --cached --others --exclude-standard
  )
  Assert-NativeSuccess "Membaca daftar file source Git"

  $SourceFiles = New-Object "System.Collections.Generic.List[string]"

  foreach ($GitFile in $GitFiles) {
    $RelativePath = ([string]$GitFile).Trim()

    if ([string]::IsNullOrWhiteSpace($RelativePath)) {
      continue
    }

    if (Test-ExcludedSourcePath $RelativePath) {
      continue
    }

    $FullPath = Join-Path -Path $GitRoot -ChildPath $RelativePath

    if (-not (Test-Path -LiteralPath $FullPath -PathType Leaf)) {
      continue
    }

    if (-not $SourceFiles.Contains($RelativePath)) {
      $SourceFiles.Add($RelativePath)
    }
  }

  if ($SourceFiles.Count -eq 0) {
    throw "Tidak ada file source yang dapat dimasukkan ke ZIP."
  }

  $RequiredFiles = @(
    ".gitattributes",
    "package.json",
    "backend/package.json",
    "frontend/package.json",
    "frontend/src/data/adapters/sqlite/sqliteProductsAdapter.js",
    "frontend/src/data/adapters/sqlite/sqliteRawMaterialsAdapter.js",
    "frontend/src/data/adapters/sqlite/sqliteSemiFinishedMaterialsAdapter.js",
    "frontend/src/data/adapters/sqlite/sqliteStockReadModelsAdapter.js",
    "frontend/src/data/adapters/sqlite/sqlitePricingRulesAdapter.js"
  )

  foreach ($RequiredFile in $RequiredFiles) {
    if (-not $SourceFiles.Contains($RequiredFile)) {
      throw "File source wajib tidak ditemukan atau tidak ikut ZIP: $RequiredFile"
    }
  }

  Add-Type -AssemblyName System.IO.Compression
  Add-Type -AssemblyName System.IO.Compression.FileSystem

  $FileStream = $null
  $ZipArchive = $null

  try {
    $FileStream = [System.IO.File]::Open(
      $OutputZip,
      [System.IO.FileMode]::CreateNew,
      [System.IO.FileAccess]::Write,
      [System.IO.FileShare]::None
    )

    $ZipArchive = [System.IO.Compression.ZipArchive]::new(
      $FileStream,
      [System.IO.Compression.ZipArchiveMode]::Create,
      $false
    )

    foreach ($RelativePath in $SourceFiles) {
      $NormalizedPath = $RelativePath.Replace("\", "/").TrimStart([char[]]@("/"))
      $FullPath = Join-Path -Path $GitRoot -ChildPath $RelativePath

      [System.IO.Compression.ZipFileExtensions]::CreateEntryFromFile(
        $ZipArchive,
        $FullPath,
        $NormalizedPath,
        [System.IO.Compression.CompressionLevel]::Optimal
      ) | Out-Null
    }
  }
  catch {
    Remove-Item -LiteralPath $OutputZip -Force -ErrorAction SilentlyContinue
    throw
  }
  finally {
    if ($null -ne $ZipArchive) {
      $ZipArchive.Dispose()
    }

    if ($null -ne $FileStream) {
      $FileStream.Dispose()
    }
  }

  if (-not (Test-Path -LiteralPath $OutputZip -PathType Leaf)) {
    throw "ZIP gagal dibuat: $OutputZip"
  }

  $ReadStream = $null
  $ReadArchive = $null

  try {
    $ReadStream = [System.IO.File]::OpenRead($OutputZip)
    $ReadArchive = [System.IO.Compression.ZipArchive]::new(
      $ReadStream,
      [System.IO.Compression.ZipArchiveMode]::Read,
      $false
    )

    $ZipEntryCount = @($ReadArchive.Entries | Where-Object {
      -not [string]::IsNullOrWhiteSpace($_.Name)
    }).Count
  }
  finally {
    if ($null -ne $ReadArchive) {
      $ReadArchive.Dispose()
    }

    if ($null -ne $ReadStream) {
      $ReadStream.Dispose()
    }
  }

  if ($ZipEntryCount -ne $SourceFiles.Count) {
    Remove-Item -LiteralPath $OutputZip -Force -ErrorAction SilentlyContinue
    throw "Jumlah file ZIP tidak sama dengan source. Source: $($SourceFiles.Count), ZIP: $ZipEntryCount."
  }

  Write-Host ""
  Write-Host "Memverifikasi ZIP..."

  & node "scripts/verify-source-ready.cjs" "--archive-only" $OutputZip
  $VerifyExitCode = $LASTEXITCODE

  if ($VerifyExitCode -ne 0) {
    Remove-Item -LiteralPath $OutputZip -Force -ErrorAction SilentlyContinue
    throw "ZIP tidak lolos verifier IMS."
  }

  $SizeMB = [math]::Round((Get-Item -LiteralPath $OutputZip).Length / 1MB, 2)
  $Sha256 = (Get-FileHash -LiteralPath $OutputZip -Algorithm SHA256).Hash.ToLowerInvariant()

  Write-Host ""
  Write-Host "ZIP selesai dibuat dan lolos verifier."
  Write-Host "Nama        : $OutputFileName"
  Write-Host "File source : $($SourceFiles.Count)"
  Write-Host "File ZIP    : $ZipEntryCount"
  Write-Host "Lokasi      : $OutputZip"
  Write-Host "Ukuran      : $SizeMB MB"
  Write-Host "SHA-256     : $Sha256"
}
